function pattern19(rows: number) {
  let output = "";
  for (let i = rows; i >= 1; i--) {
    for (let j = 1; j < rows*2; j++) {
      output += Math.abs(rows - i) > Math.abs(rows - j) ? " " : "*";
    }
    output += "\n";
  }
  for (let i = 1; i <= rows; i++) {
    for (let j = 1; j < rows*2; j++) {
      output += Math.abs(rows - i) > Math.abs(rows - j) ? " " : "*";
    }
    output += "\n";
  }
  
  console.log(output);
}
